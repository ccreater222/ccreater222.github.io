            $this->Filename = $_GET['name'];
        }

            $this->Filename = $sandbox."admin".$ext;

        global $sandbox;
        global $ext;

            $this->cmd = "die('illegal filename!');";
        }
        else{
            if($file['size'] > 1024){

            }
        }
    }
